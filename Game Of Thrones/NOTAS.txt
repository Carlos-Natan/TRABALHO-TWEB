REGRAS:

    RESPONSIVIDADE[ ->(nessa eu não sei oq fazer então não mexi)

        rem: tamanhos de fonte / espaçamentos (padding/margin) / tamanhos mínimos de elementos / bordas, sombras, gaps, parágrafos, menus, botões, subtítulos e textos gerais
             1rem = 16px | 1.5rem = 24px | 2rem = 32px
        %: larguras internas.
        vw: títulos / banners / imagens de fundo que preenchem a tela / textos grandes e decorativos que precisam acompanhar a largura da tela.
        fr: grid de cards.

        EXEMPLO:

            main {
                max-width: 900px;
                margin: auto;
                padding: 2rem;
            }

            h1 {
                font-size: 3rem;
            }

            .hero {
                height: 40vw;
                background: url(IMG/capa.jpg) center/cover;
            }

            .cards {
                display: grid;
                grid-template-columns: repeat(auto-fit, minmax(250px, 1fr));
                gap: 2rem;
            }

    ]
